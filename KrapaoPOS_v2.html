📊 เชื่อมต่อ Google Sheets
วิธีตั้งค่า:
1. เปิด Google Sheets → Extensions → Apps Script
2. วาง script จาก bit.ly/sheets-webhook
3. Deploy → Web App → Copy URL มาวางด้านล่าง
Google Apps Script Web App URL
Apps Script สำหรับวาง:
function doPost(e){ var data=JSON.parse(e.postData.contents); var ss=SpreadsheetApp.getActiveSpreadsheet(); var sheetName=data.sheet||"Sales"; var sheet=ss.getSheetByName(sheetName)||ss.insertSheet(sheetName); if(sheetName==="Sales"&&sheet.getLastRow()===0) sheet.appendRow(["ID","วันที่","เวลา","รายการ","ยอด","ส่วนลด","ชำระ","พนักงาน"]); if(sheetName==="Expenses"&&sheet.getLastRow()===0) sheet.appendRow(["ID","วันที่","เวลา","หมวด","รายการ","จำนวน","พนักงาน"]); if(sheetName==="Sales") sheet.appendRow([data.id,data.date,data.time,data.items,data.total,data.discount,data.method,data.staff]); if(sheetName==="Expenses") sheet.appendRow([data.id,data.date,data.time,data.category,data.name,data.amount,data.staff]); return ContentService.createTextOutput("OK"); }
